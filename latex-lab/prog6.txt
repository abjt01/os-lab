% 4. Animations (Unveiling Content)
\documentclass{beamer}
\usetheme{Madrid} % Choose a theme
\title{My Presentation Title}
\title{LateX documentation}
\author{Sanjana M Nagaraj}
\date{\today}

\begin{document}
	
	\frame{\titlepage} % Title frame
\begin{frame}
	\frametitle{Step-by-Step Reveal}
	\begin{itemize}
		\item<1-> First point (appears on slide 1)
		\item<2-> Second point (appears on slide 2)
		\item<3> Third point (appears only on slide 3)
	\end{itemize}
	
	\only<4>{This text appears only on slide 4}
\end{frame}
\end{document}

	\frame{\titlepage} % Title frame

	\begin{figure}
		\centering
		\includegraphics[width=0.3\textwidth]{image1.png}
		\caption{An example figure}
	\end{figure}
\end{frame}

\begin{frame}
	\frametitle{Table Example}
	\begin{table}
		\centering
		\begin{tabular}{|c|c|}
			\hline
			Header 1 & Header 2 \\ \hline
			Data 1 & Data 2 \\ \hline
		\end{tabular}
		\caption{An example table}
	\end{table}
\end{frame}
\end{document}
