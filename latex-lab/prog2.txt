\subsection*{Standard and Custom Paper Sizes}

\section*{Letter (8.5 $\times$ 11 in)}
\lipsum[1]

\clearpage

% A4 size
\newgeometry{paper=a4paper, margin=1in}
\section*{A4 (210 $\times$ 297 mm)}
This page uses A4 paper size, common in many countries.
\lipsum[2]

\clearpage

% Legal size
\newgeometry{paper=legalpaper, margin=1in}
\section*{Legal (8.5 $\times$ 14 in)}
This page uses Legal paper size.
\lipsum[3]

\clearpage

% Custom postcard size
\newgeometry{paperwidth=100mm, paperheight=150mm, margin=10mm}
\section*{Custom Postcard (100 $\times$ 150 mm)}
This mimics a postcard-style page.

\clearpage

% Square book-style page
\newgeometry{paperwidth=6in, paperheight=6in, margin=0.75in}
\section*{Square Format Page (6 $\times$ 6 in)}
This simulates a square book layout.

\clearpage

% Restore normal layout
\restoregeometry
